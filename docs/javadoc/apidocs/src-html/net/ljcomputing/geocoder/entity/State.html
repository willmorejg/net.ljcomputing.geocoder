032package net.ljcomputing.geocoder.entity;
033
034import java.util.HashMap;
035import java.util.Map;
036
037/**
038 * Enumeration of states in the United States.
039 * 
040 * Modified from <a target="_blank" href="https://gist.github.com/webdevwilson/5271984">https://gist.github.com/webdevwilson/5271984</a> (created by Kerry Wilson).
041 * 
042 * @author James G. Willmore
043 */
044public enum State {
045        ALABAMA("Alabama", "AL"), 
046        ALASKA("Alaska", "AK"), 
047        AMERICAN_SAMOA("American Samoa", "AS"), 
048        ARIZONA("Arizona", "AZ"), 
049        ARKANSAS("Arkansas", "AR"), 
050        CALIFORNIA("California", "CA"), 
051        COLORADO("Colorado", "CO"), 
052        CONNECTICUT("Connecticut", "CT"), 
053        DELAWARE("Delaware", "DE"), 
054        DISTRICT_OF_COLUMBIA("District of Columbia", "DC"), 
055        FEDERATED_STATES_OF_MICRONESIA("Federated States of Micronesia", "FM"), 
056        FLORIDA("Florida", "FL"), 
057        GEORGIA("Georgia", "GA"), 
058        GUAM("Guam", "GU"), 
059        HAWAII("Hawaii","HI"), 
060        IDAHO("Idaho", "ID"), 
061        ILLINOIS("Illinois", "IL"), 
062        INDIANA("Indiana", "IN"), 
063        IOWA("Iowa", "IA"), 
064        KANSAS("Kansas", "KS"), 
065        KENTUCKY("Kentucky", "KY"), 
066        LOUISIANA("Louisiana", "LA"), 
067        MAINE("Maine","ME"), 
068        MARYLAND("Maryland", "MD"), 
069        MARSHALL_ISLANDS("Marshall Islands", "MH"), 
070        MASSACHUSETTS("Massachusetts", "MA"), 
071        MICHIGAN("Michigan", "MI"), 
072        MINNESOTA("Minnesota", "MN"), 
073        MISSISSIPPI("Mississippi", "MS"), 
074        MISSOURI("Missouri", "MO"), 
075        MONTANA("Montana", "MT"), 
076        NEBRASKA("Nebraska", "NE"), 
077        NEVADA("Nevada", "NV"), 
078        NEW_HAMPSHIRE("New Hampshire", "NH"), 
079        NEW_JERSEY("New Jersey", "NJ"), 
080        NEW_MEXICO("New Mexico", "NM"), 
081        NEW_YORK("New York", "NY"), 
082        NORTH_CAROLINA("North Carolina", "NC"), 
083        NORTH_DAKOTA("North Dakota", "ND"), 
084        NORTHERN_MARIANA_ISLANDS("Northern Mariana Islands", "MP"), 
085        OHIO("Ohio", "OH"), 
086        OKLAHOMA("Oklahoma", "OK"), 
087        OREGON("Oregon", "OR"), 
088        PALAU("Palau","PW"), 
089        PENNSYLVANIA("Pennsylvania", "PA"), 
090        PUERTO_RICO("Puerto Rico", "PR"), 
091        RHODE_ISLAND("Rhode Island", "RI"), 
092        SOUTH_CAROLINA("South Carolina", "SC"), 
093        SOUTH_DAKOTA("South Dakota", "SD"), 
094        TENNESSEE("Tennessee", "TN"), 
095        TEXAS("Texas", "TX"), 
096        UTAH("Utah", "UT"), 
097        VERMONT("Vermont", "VT"), 
098        VIRGIN_ISLANDS("Virgin Islands", "VI"), 
099        VIRGINIA("Virginia", "VA"), 
100        WASHINGTON("Washington", "WA"), 
101        WEST_VIRGINIA("West Virginia","WV"), 
102        WISCONSIN("Wisconsin", "WI"), 
103        WYOMING("Wyoming", "WY"), 
104        UNKNOWN("Unknown", "");
105
106  /**
107   * The state's name.
108   */
109  private String name;
110
111  /**
112   * The state's abbreviation.
113   */
114  private String abbreviation;
115
116  /**
117   * The set of states addressed by abbreviations.
118   */
119  private static final Map<String, State> STATES_BY_ABBR = new HashMap<String, State>();
120
121  /* static initializer */
122  static {
123    for (State state : values()) {
124      STATES_BY_ABBR.put(state.getAbbreviation(), state);
125    }
126  }
127
128  /**
129   * Constructs a new state.
130   * 
131   * @param name
132   *          the state's name.
133   * @param abbreviation
134   *          the state's abbreviation.
135   */
136  State(final String name, final String abbreviation) {
137    this.name = name;
138    this.abbreviation = abbreviation;
139  }
140
141  /**
142   * Returns the state's name.
143   * 
144   * @return the state's name.
145   */
146  public String getName() {
147    return name;
148  }
149
150  /**
151   * Returns the state's abbreviation.
152   * 
153   * @return the state's abbreviation.
154   */
155  public String getAbbreviation() {
156    return abbreviation;
157  }
158
159  /**
160   * Gets the enum constant with the specified abbreviation.
161   *
162   * @param abbr
163   *          the state's abbreviation.
164   * @return the enum constant with the specified abbreviation.
165   */
166  public static State valueOfAbbreviation(final String abbr) {
167    final State state = STATES_BY_ABBR.get(abbr);
168    if (state != null) {
169      return state;
170    } else {
171      return UNKNOWN;
172    }
173  }
174
175  /**
176   * State value of given name.
177   *
178   * @param name
179   *          the name
180   * @return the state
181   */
182  public static State valueOfName(final String name) {
183    final String enumName = name.toUpperCase().replaceAll(" ", "_");
184    try {
185      return valueOf(enumName);
186    } catch (final IllegalArgumentException e) {
187      return State.UNKNOWN;
188    }
189  }
